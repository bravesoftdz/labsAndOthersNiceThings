function print_menu()
{
global $page_header, $page_footer;
echo "
$page_header
<p>
<form action='index.php' method='post'>
Title: <input type='text' name='title'>
<br>
E-mail: <input type='text' name='email'>
<input type='hidden' name='tp' value='submit'>
<br>
<input type='submit'>
</form>
<p>
<a href='index.php?tp=view'>View Entries</a>
$page_footer
";
exit;
}

function submit_entry()
{
global $page_header, $page_footer, $db_table, $email, $title;

$sql="insert into $db_table ( title, email ) values( '$title', '$email')";
mysql_query("$sql");

echo "
$page_header
<p> 
Entry has been submitted
$page_footer
";
exit;
}

function view_entries()
{
global $page_header, $page_footer, $db_table;

echo "
$page_header
";

{
echo "<p>
".$row['id'].".<br>
title: ".$row['title']."
; 
<br> e-mail: ".$row['email']."; 
";
}

echo "
$page_footer
";
exit;
}
